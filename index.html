using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.IO;
using System.Net.Http.Headers;
            
namespace AutoLibrary
{
    public class auto
    {
        public int Id;
        public string Marka;
        public string Modell;
        public int Gyártásiev;
        public string Szin;
        public int EladottDb;
        public int AvgEladasAr;
            
        public auto(int id, string marka, string modell, int gyártásiev, string szin, int eladottDb, int avgEladasAr)
        {
            Id = id;
            Marka = marka;
            Modell = modell;
            Gyártásiev = gyártásiev;
            Szin = szin;
            EladottDb = eladottDb;
            AvgEladasAr = avgEladasAr;
        }
            
        public int Id1 { get => Id; set => Id = value; }
        public string Marka1 { get => Marka; set => Marka = value; }
        public string Modell1 { get => Modell; set => Modell = value; }
        public int Gyártásiev1 { get => Gyártásiev; set => Gyártásiev = value; }
        public string Szin1 { get => Szin; set => Szin = value; }
        public int EladottDb1 { get => EladottDb; set => EladottDb = value; }
        public int AvgEladasAr1 { get => AvgEladasAr; set => AvgEladasAr = value; }
            
            
            
        public static List<auto> OpenFiles(string filenev)
        {
            List<auto> autoList = new List<auto>();
            
            if (!File.Exists(filenev))
                return autoList;
            
            string[] sorok = File.ReadAllLines(filenev);
            
            for (int i = 1; i < sorok.Length; i++) // fejléct kihagyva
            {
                string[] mezok = sorok[i].Split(';');
            
                if (mezok.Length != 7)
                    continue; // hibás sor: nem 7 mező
            
                if (
                    int.TryParse(mezok[0], out int id) &&
                    int.TryParse(mezok[3], out int ev) &&
                    int.TryParse(mezok[5], out int eladasDb) &&
                    int.TryParse(mezok[6], out int ar)
                )
                {
                    auto a = new auto(id, mezok[1], mezok[2], ev, mezok[4], eladasDb, ar);
                    autoList.Add(a);
                }
            }
            
            return autoList;
        }
    }
}

let nextId = 1;

function rendeles(event) {
    event.preventDefault();
    
    let inputvezeteknev = document.querySelector("input[name='vezeteknev']");
    let inputkeresztnev = document.querySelector("input[name='keresztnev']");
    let inputemailcim = document.querySelector("input[name='email']");
    let inputtelefonszam = document.querySelector("input[name='telefonszam']");
    
    let megrendeloError = document.getElementById("megrendeloError");
    if (!megrendeloError) {
        megrendeloError = document.createElement("div");
        megrendeloError.id = "megrendeloError";
        megrendeloError.style.color = "red";
        document.getElementById("userForm").appendChild(megrendeloError);
    }

    if(inputvezeteknev.value == "" || inputkeresztnev.value == "" || inputemailcim.value == "" || inputtelefonszam.value == "") {
        megrendeloError.innerHTML = "Hiányzó felhasználói adatok!";
    } 
    else {
        const tableBody = document.querySelector("tbody");
        
        const newRow = document.createElement("tr");
        
        newRow.setAttribute("data-id", nextId);
        
        newRow.innerHTML = `
            <td>${nextId}</td>
            <td>${inputvezeteknev.value}</td>
            <td>${inputkeresztnev.value}</td>
            <td>${inputemailcim.value}</td>
            <td>${inputtelefonszam.value}</td>
        `;
        
        tableBody.appendChild(newRow);
        
        nextId++;

        document.getElementById("userForm").reset();
        
        megrendeloError.innerHTML = "";
    }
}
document.addEventListener("DOMContentLoaded", function() {
    const form = document.getElementById("userForm");
    form.addEventListener("submit", rendeles);
    
    const headerRow = document.querySelector("thead tr");
    const idHeader = document.createElement("th");
    idHeader.textContent = "ID";
    headerRow.insertBefore(idHeader, headerRow.firstChild);
    
    const tableBody = document.querySelector("tbody");
    tableBody.innerHTML = "";
});

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.IO;

namespace Maraton
{
    class adatok
    {
        string Rajtszam;
        string Kategoria;
        string Nev;
        string Egyesulet;
        TimeSpan Ido;

        public adatok(string rajtszam, string kategoria, string nev, string egyesulet, TimeSpan ido)
        {
            Rajtszam = rajtszam;
            Kategoria = kategoria;
            Nev = nev;
            Egyesulet = egyesulet;
            Ido = ido;
        }

        public string Rajtszam1 { get => Rajtszam; set => Rajtszam = value; }
        public string Kategoria1 { get => Kategoria; set => Kategoria = value; }
        public string Nev1 { get => Nev; set => Nev = value; }
        public string Egyesulet1 { get => Egyesulet; set => Egyesulet = value; }
        public TimeSpan Ido1 { get => Ido; set => Ido = value; }

        public List<adatok> OpenFiles(string filenev)
        {
            List<adatok> maratonList = new List<adatok>();
            string[] sorok = File.ReadAllLines(filenev);
            for (int i = 1; i < sorok.Length; i++)
            {
                string[] mezok = sorok[i].Split(';');
                //auto a = new auto(0, "", "", 0, "", 0, 0);
                adatok a = new adatok(mezok[0], mezok[1], mezok[2], mezok[3], TimeSpan.Parse(mezok[4]));
                maratonList.Add(a);
            }
            return maratonList;
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Maraton
{
    class Versenytav
    {
        private string Rajtszam;
        public string Tav
        {
            get
            {
                switch (Rajtszam[0])
                {
                    case 'M': return "Mini";
                    case 'R': return "Rövid";
                    case 'K': return "Közép";
                    case 'H': return "Hosszú";
                    case 'E': return "Pedelec";
                }
                return "Hibás rajtszám";
            }
        }
        public Versenytav(string rajtszam)
        {
            Rajtszam = rajtszam;
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.IO;

namespace Maraton
{
    class Program
    {
        static void Main(string[] args)
        {
            
            adatok a = new adatok("", "", "", "", TimeSpan.Zero);
            List<adatok> versenyzok = a.OpenFiles("bukkm2019.txt");
            int osszesIndulo = 691;
            int celbaErkezettek = versenyzok.Count;
            double NemTeljesítők = osszesIndulo - celbaErkezettek;
            double szazalek = NemTeljesítők / osszesIndulo * 100;
            Console.WriteLine($"4. feladat: Versenytávot nem teljesítők: {szazalek}%");

            int rovidtavNok = versenyzok.Count(v =>
            {
                Versenytav tav = new Versenytav(v.Rajtszam1);
                return tav.Tav == "Rövid" && v.Kategoria1.EndsWith("n");
            });
            Console.WriteLine($"5. feladat: Rövidtávú női versenyzők száma: {rovidtavNok} fő");

            bool voltIlyen = versenyzok.Any(v => v.Ido1 > TimeSpan.FromHours(6));
            if (voltIlyen)
            {
                Console.WriteLine("6. feladat: Volt ilyen versenyző");
            }
            else
            {
                Console.WriteLine("6. feladat: Nem volt ilyen versenyző");
            }

            var rovidFelnottFerfiak = versenyzok
            .Where(v => new Versenytav(v.Rajtszam1).Tav == "Rövid" && v.Kategoria1 == "ff")
            .ToList();

            var gyoztes = rovidFelnottFerfiak
                .OrderBy(v => v.Ido1)
                .First();

            Console.WriteLine("7. feladat: A felnőtt férfi (ff) kategória győztese rövidtávon:");
            Console.WriteLine($"\tRajtszám: {gyoztes.Rajtszam1}");
            Console.WriteLine($"\tNév: {gyoztes.Nev1}");
            if (!string.IsNullOrWhiteSpace(gyoztes.Egyesulet1))
            {
                Console.WriteLine($"\tEgyesület: {gyoztes.Egyesulet1}");
            }
            Console.WriteLine($"\tIdő: {gyoztes.Ido1}");

            var ferfiVersenyzok = versenyzok
            .Where(v => v.Kategoria1.EndsWith("f"));

            var statisztika = ferfiVersenyzok
                .GroupBy(v => v.Kategoria1)
                .ToDictionary(g => g.Key, g => g.Count());

            Console.WriteLine("8. feladat: Férfi versenyzők statisztikája kategóriánként:");
            foreach (var sor in statisztika)
            {
                Console.WriteLine($"\t{sor.Key} - {sor.Value} fő");
            }
        }
    }
